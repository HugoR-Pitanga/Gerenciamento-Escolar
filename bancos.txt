create table aluno(
   id serial,
   nome varchar(255) not null,
   primary key (id)
);

create table nota(
   nota_id serial,
   materia varchar(255) not null,
   nota float not null,
   aluno_id int,
   primary key (nota_id),
   foreign key (aluno_id) references aluno (id)
);

create table token(
   token_id serial,
   token_user varchar(255) not null,
   token_token varchar(255) not null, 
   data_valida datetime not null
);
